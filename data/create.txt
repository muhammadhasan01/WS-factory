user (id, username, email, password)
coklat (id, nama, jumlah)
bahan (id, nama, jumlah, kedaluwarsa)
resep (id_coklat, bahan, jumlah, harga)  *tipe bahan dan jumlah adalah array of int
request_add_stock (id, id_coklat, jumlah, status)
saldo (id, jumlah)

create table user (
    id int primary key auto_increment,
    username varchar(255) not null,
    password varchar(255) not null
);

create table coklat (
    id int primary key auto_increment,
    nama varchar(256) not null unique,
    jumlah int not null default 0
);

create table request_add_stock (
    id int primary key auto_increment,
    id_coklat int not null,
    jumlah int,
    status varchar(255) default 'PENDING',
    foreign key (id_coklat) references coklat(id)
);

create table saldo (
    amount int not null,
    CHECK (amount >= 0)
);

create table bahan (
    id int primary key auto_increment,
    nama varchar(255) not null,
    jumlah int not null,
    kedaluarsa date
);

create table resep(
    id int primary key auto_increment,
    id_coklat int,
    nama_bahan varchar(255) not null,
    jumlah int,
    harga int,
    foreign key (id_coklat) references coklat(id)
);
